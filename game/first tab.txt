    <script>
        const input = document.getElementById('userInput');
        const card = document.getElementById('card');
        const frame = document.getElementById('frame');
        const feedback = document.getElementById('feedback');
        const pName = document.getElementById('p-name');
        const enterBtn = document.getElementById('enter-btn');

        const imgN = document.getElementById('img-neutral');
        const imgH = document.getElementById('img-happy');
        const imgS = document.getElementById('img-sad');

        // Logic keyword lists
        const goodWords = ["smart", "kind", "great", "nice", "hero", "good", "cool", "phelopateer", "win"];
        const badWords = ["bad", "stupid", "mean", "ugly", "hate", "worse", "fail", "rude"];

        input.addEventListener('input', (e) => {
            const val = e.target.value.toLowerCase();

            // Neutral/Empty reset
            if (val.trim() === "") {
                updateUI('neutral');
                return;
            }

            const isGood = goodWords.some(word => val.includes(word));
            const isBad = badWords.some(word => val.includes(word));

            if (isGood) {
                updateUI('happy');
            } else if (isBad) {
                updateUI('sad');
            } else {
                updateUI('neutral');
            }
        });

        function updateUI(state) {
            // Reset all visual states
            [imgN, imgH, imgS].forEach(i => i.classList.remove('active-img'));
            card.classList.remove('shake');
            card.style.borderColor = "#1a73e8";
            frame.style.borderColor = "#eaeef2";
            pName.style.display = "none";
            feedback.innerText = "";
            enterBtn.style.display = "none";

            if (state === 'happy') {
                imgH.classList.add('active-img');
                card.style.borderColor = "#4CAF50";
                frame.style.borderColor = "#4CAF50";
                pName.style.display = "block";
                feedback.innerText = "Kind people always win people heart";
                feedback.style.color = "#2e7d32";
                enterBtn.style.display = "block";
            } 
            else if (state === 'sad') {
                imgS.classList.add('active-img');
                card.style.borderColor = "#f44336";
                frame.style.borderColor = "#f44336";
                feedback.innerText = "Tell him something nicely";
                feedback.style.color = "#d32f2f";
                // Trigger the shake
                void card.offsetWidth; 
                card.classList.add('shake');
            } 
            else {
                imgN.classList.add('active-img');
            }
        }
    </script>
